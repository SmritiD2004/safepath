0. The Strategy: What Exactly Changes
SafePath today is a single-tenant consumer app â€” one user, their scenarios, their certificate. The enterprise pivot adds a multi-tenant layer on top of the existing engine without rewriting it. Your five game modes, ScenarioCore, scoring engine, AI coach, and certificate system all stay as-is. What gets added sits around them.
Core principle: the enterprise layer is a wrapper, not a replacement. All existing user-facing features continue working. New enterprise features are additive.

// lib/ai/industryPrompts.ts
export function buildEnterpriseSystemPrompt(
  industry: string,
  jobRole: string,
  complianceFramework?: string
): string {
  const industryContext: Record<string, string> = {
    CORPORATE_IT: `You are a corporate compliance coach for IT professionals.
      Focus on POSH Act, workplace ethics, and professional conduct.`,
    MANUFACTURING: `You are a factory safety coach.
      Focus on machine safety, PPE, emergency procedures, chemical hazards.
      Cite Factories Act 1948 where relevant.`,
    HEALTHCARE: `You are a clinical training coach.
      Focus on patient safety, infection control, medical ethics, triage.`,
    BANKING_FINANCE: `You are a financial compliance coach.
      Focus on fraud detection, phishing, RBI regulations, insider threats.`,
  }
  const base = industryContext[industry] ??
    'You are a workplace safety and compliance coach.'
  return `${base}
  Employee role: ${jobRole}.
  ${complianceFramework ? `Compliance framework: ${complianceFramework}.` : ''}
  Give concise, actionable feedback (max 120 words).
  Never victim-blame. Be empowering and practical.`
}

4.3  app/api/org/[orgId]/analytics/route.ts
This is the data engine for the manager dashboard. It aggregates ScenarioRun and RolePlaySession data for all OrgMembers:
export async function GET(req, { params }) {
  const { orgId } = await params
  // 1. Get all userIds in this org
  const members = await db.orgMember.findMany({
    where: { orgId },
    select: { userId: true, jobRole: true, dept: { select: { name: true } } }
  })
  const userIds = members.map(m => m.userId)
  // 2. Aggregate ScenarioRuns for those users
  const runs = await db.scenarioRun.findMany({
    where: { userId: { in: userIds }, status: 'COMPLETED' },
    select: { userId: true, scenarioId: true,
              finalConfidence: true, finalEq: true, finalRisk: true }
  })
  // 3. Build per-user summary
  const byUser = userIds.map(uid => {
    const userRuns = runs.filter(r => r.userId === uid)
    return {
      userId: uid,
      completedScenarios: userRuns.length,
      avgConfidence: avg(userRuns.map(r => r.finalConfidence ?? 50)),
      avgEq:         avg(userRuns.map(r => r.finalEq ?? 50)),
      avgRisk:       avg(userRuns.map(r => r.finalRisk ?? 40)),
    }
  })
  return NextResponse.json({
    totalMembers: members.length,
    completionRate: pct(byUser.filter(u => u.completedScenarios > 0).length, members.length),
    avgOrgConfidence: avg(byUser.map(u => u.avgConfidence)),
    memberStats: byUser,
  })
}
Helper: function avg(nums) { return nums.length ? Math.round(nums.reduce((a,b)=>a+b,0)/nums.length) : 0 }
5. Compliance Reports  (app/api/org/[orgId]/reports/)
Companies need audit-ready records. Generate a CSV report of all employee training completions for a time period, downloadable from the manager dashboard.
5.1  POST /api/org/[orgId]/reports â€” generate CSV
import { stringify } from 'csv-stringify/sync'  // npm install csv-stringify
import { writeFile } from 'fs/promises'
import path from 'path'
export async function POST(req, { params }) {
  const { orgId } = await params
  const { period } = await req.json()  // e.g. '2025-Q1'
  const members = await db.orgMember.findMany({
    where: { orgId },
    include: { user: { select: { name: true, email: true } },
               dept: { select: { name: true } } }
  })
  const rows = await Promise.all(members.map(async m => {
    const cert = await db.userCertificate.findUnique({
      where: { userId: m.userId }
    })
    const runs = await db.scenarioRun.count({
      where: { userId: m.userId, status: 'COMPLETED' }
    })
    return {
      Name:         m.user.name ?? 'Unknown',
      Email:        m.user.email ?? '',
      Department:   m.dept?.name ?? 'Unassigned',
      Role:         m.jobRole ?? '',
      'Scenarios Completed': runs,
      'Certificate Issued':  cert ? 'Yes' : 'No',
      'Certificate Date':    cert?.issuedAt?.toISOString().split('T')[0] ?? '',
      'Cert Code':           cert?.code ?? '',
    }
  }))
  const csv = stringify(rows, { header: true })
  const filename = `compliance-${orgId}-${period}-${Date.now()}.csv`
  const filepath = path.join(process.cwd(), 'public', 'reports', filename)
  await writeFile(filepath, csv)
  await db.complianceReport.create({
    data: { orgId, period, csvUrl: `/reports/${filename}` }
  })
  return NextResponse.json({ csvUrl: `/reports/${filename}` })
}

7. AI Personalisation by Job Role
When an employee logs in through an org context, their job role and department risk profile determine which scenarios are recommended first and how the AI coach frames feedback.
7.1  Create lib/enterprise/personalisation.ts
// lib/enterprise/personalisation.ts
interface PersonalisationContext {
  industry:    string
  jobRole:     string
  deptRisk:    string  // e.g. 'high-machinery'
  packageIds:  string[]
}
export function getRecommendedScenarios(
  ctx: PersonalisationContext,
  allPackages: typeof ENTERPRISE_PACKAGES
): string[] {
  // Filter packages for this employee's org
  const orgPackages = ctx.packageIds
    .map(id => allPackages[id])
    .filter(Boolean)
  // Prioritise scenarios matching job role keywords
  const roleKeywords = ctx.jobRole.toLowerCase().split(' ')
  const scored = orgPackages.flatMap(p => p.scenarioIds).map(sid => ({
    sid,
    score: roleKeywords.filter(kw => sid.includes(kw)).length
  }))
  return scored.sort((a,b) => b.score - a.score).map(s => s.sid)
}
7.2  Use in /api/org/[orgId]/recommended-scenarios
export async function GET(req, { params }) {
  const session = await auth()
  const { orgId } = await params
  const member = await db.orgMember.findUnique({
    where: { userId_orgId: { userId: session.user.id, orgId } },
    include: { org: { include: { packages: true } }, dept: true }
  })
  const scenarioIds = getRecommendedScenarios({
    industry:   member.org.industry,
    jobRole:    member.jobRole ?? 'Employee',
    deptRisk:   member.dept?.riskProfile ?? '',
    packageIds: member.org.packages.map(p => p.packageId),
  }, ENTERPRISE_PACKAGES)
  return NextResponse.json({ scenarioIds })
}
